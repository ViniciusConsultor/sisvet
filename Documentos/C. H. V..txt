-- Table: cliente

-- DROP TABLE cliente;

CREATE TABLE cliente
(
  cod_cliente serial NOT NULL,
  nome_cliente character varying(50) NOT NULL,
  data_nascimento_cliente date NOT NULL,
  "endereço_cliente" character varying(100) NOT NULL,
  telefone_cliente character varying(13) NOT NULL,
  rg_cliente character varying(15) NOT NULL,
  expedidor_rg_cliente character(5) NOT NULL,
  cpf_cliente character(14) NOT NULL,
  municipio_cliente character varying(50) NOT NULL,
  CONSTRAINT cod_cliente_pk PRIMARY KEY (cod_cliente ),
  CONSTRAINT cpf_cliente UNIQUE (cpf_cliente )
)
WITH (
  OIDS=FALSE
);
ALTER TABLE cliente
  OWNER TO postgres;







-- Table: pagamento

-- DROP TABLE pagamento;

CREATE TABLE pagamento
(
  valor_pagamento numeric(8,2),
  cod_pagamento serial NOT NULL,
  CONSTRAINT cod_pagamento_pk PRIMARY KEY (cod_pagamento )
)
WITH (
  OIDS=FALSE
);
ALTER TABLE pagamento
  OWNER TO postgres;





-- Table: paciente

-- DROP TABLE paciente;

CREATE TABLE paciente
(
  cod_paciente serial NOT NULL,
  rghv character varying(10) NOT NULL,
  nome_paciente character varying,
  especie_paciente character varying(50) NOT NULL,
  raca_paciete character varying(50) NOT NULL,
  nascimento_paciente date,
  pelagem_paciente character varying(50) NOT NULL,
  sexo_paciente character(1) NOT NULL,
  castrado_paciente character(1),
  CONSTRAINT cod_paciente_pk PRIMARY KEY (cod_paciente )
)
WITH (
  OIDS=FALSE
);
