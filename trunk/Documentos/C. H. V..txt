-- Table: cliente

-- DROP TABLE cliente;

CREATE TABLE cliente
(
  cod_cliente serial NOT NULL,
  nome_cliente character varying(50) NOT NULL,
  data_nascimento_cliente date NOT NULL,
  "endereço_cliente" character varying(100) NOT NULL,
  telefone_cliente character varying(13) NOT NULL,
  rg_cliente character varying(15) NOT NULL,
  expedidor_rg_cliente character(5) NOT NULL,
  cpf_cliente character(14) NOT NULL,
  municipio_cliente character varying(50) NOT NULL,
  CONSTRAINT cod_cliente_pk PRIMARY KEY (cod_cliente ),
  CONSTRAINT cpf_cliente UNIQUE (cpf_cliente )
)
WITH (
  OIDS=FALSE
);
ALTER TABLE cliente
  OWNER TO postgres;

-- Table: medico_auxiliar

-- DROP TABLE medico_auxiliar;

CREATE TABLE medico_auxiliar
(
  auxiliarescod_auxiliar integer NOT NULL,
  medico_veterinariocod_medico_vet integer NOT NULL,
  CONSTRAINT medico_auxiliar_pkey PRIMARY KEY (auxiliarescod_auxiliar , medico_veterinariocod_medico_vet ),
  CONSTRAINT auxiliarescod_auxiliar_fk FOREIGN KEY (auxiliarescod_auxiliar)
      REFERENCES auxiliares (cod_auxiliar) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT medico_veterinariocod_medico_vet_fk FOREIGN KEY (medico_veterinariocod_medico_vet)
      REFERENCES medico_veterinario (cod_medico_vet) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION

  OIDS=FALSE
);

  OWNER TO postgres;
